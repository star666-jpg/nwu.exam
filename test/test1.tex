	\begin{question}[10]
		求函数 $f(x) = x^3 - 3x$ 的极值。
		\sol % 自动生成 "解："
		求导可得 $f'(x) = 3x^2 - 3$。
		
		令 $f'(x)=0$，解得 $x=\pm 1$。
		
		\vspace{2cm} % 留出做题空间
		
		所以，极大值为 $f(-1)=2$，极小值为 $f(1)=-2$。
	\end{question}

	\begin{question}[12]
		已知 $a > 0, b > 0$，求证：$\frac{a+b}{2} \ge \sqrt{ab}$。
		\zm % 自动生成 "证明："
		因为 $(\sqrt{a} - \sqrt{b})^2 \ge 0$，
		
		展开得 $a - 2\sqrt{ab} + b \ge 0$，
		
		移项得 $a + b \ge 2\sqrt{ab}$，
		
		所以 $\frac{a+b}{2} \ge \sqrt{ab}$。
	\end{question}
